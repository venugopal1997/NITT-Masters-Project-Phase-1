def chsh_game(strategy):
    # Referee chooses x and y randomly
    x1, x2, x3, x4, x5 = randint(0, 2), randint(0, 2), randint(0, 2), randint(0, 2), randint(0, 2)

    # Use strategy to choose a and b
    a1, a2, a3, a4, a5 = strategy(x1, x2, x3, x4, x5)
    
    x1, x2, x3, x4, x5 = int(x1), int(x2), int(x3), int(x4), int(x5)
    a1, a2, a3, a4, a5 = int(a1), int(a2), int(a3), int(a4), int(a5)
    nx1, nx2, nx3, nx4, nx5 = 1 if (x1==0) else 0, 1 if (x2==0) else 0, 1 if (x3==0) else 0, 1 if (x4==0) else 0, 1 if (x5==0) else 0
    na1, na2, na3, na4, na5 = 1 if (a1==0) else 0, 1 if (a2==0) else 0, 1 if (a3==0) else 0, 1 if (a4==0) else 0, 1 if (a5==0) else 0

    lhs = (x1 and x2 and x3) or (x1 and x2 and x4) or (x1 and x2 and x5) or (x1 and x3 and x4) or (x1 and x3 and x5) or (x1 and x4 and x5) or (x2 and x3 and x4) or (x2 and x3 and x5) or (x2 and x4 and x5) or (x3 and x4 and x5)
    rhs = a1^a2^a3^a4^a5

    # Referee decides if Alice and Bob win or lose
    if (lhs == rhs):
        return 1  # Win
    return 0  # Lose

def chsh_circuit(x1, x2, x3, x4, x5):
    N_qubits=5
    qc = QuantumCircuit(N_qubits, N_qubits)
    qc.h(0)
    for i in range (0, N_qubits-1):
        qc.cx(i, i+1)
    qc.barrier()

    a1=t1
    a2=t2
    a3=t3
    a4=a1
    a5=a2+16
    a6=a3+16
    

    # Player_1
    if x1 == 0:
        qc.rx((a1*pi) / 2, 0)
        qc.ry((a2*pi) / 32, 0)
        qc.rz((a3*pi) / 32, 0)
    else:
        qc.rx((a4*pi) / 2, 0)
        qc.ry((a5*pi) / 32, 0)
        qc.rz((a6*pi) / 32, 0)


    # Player_2
    if x2 == 0:
        qc.rx((a1*pi) / 2, 1)
        qc.ry((a2*pi) / 32, 1)
        qc.rz((a3*pi) / 32, 1)
    else:
        qc.rx((a4*pi) / 2, 1)
        qc.ry((a5*pi) / 32, 1)
        qc.rz((a6*pi) / 32, 1)


    # Player_3
    if x3 == 0:
        qc.rx((a1*pi) / 2, 2)
        qc.ry((a2*pi) / 32, 2)
        qc.rz((a3*pi) / 32, 2)
    else:
        qc.rx((a4*pi) / 2, 2)
        qc.ry((a5*pi) / 32, 2)
        qc.rz((a6*pi) / 32, 2)

    # Player_4
    if x4 == 0:
        qc.rx((a1*pi) / 2, 3)
        qc.ry((a2*pi) / 32, 3)
        qc.rz((a3*pi) / 32, 3)
    else:
        qc.rx((a4*pi) / 2, 3)
        qc.ry((a5*pi) / 32, 3)
        qc.rz((a6*pi) / 32, 3)

    # Player_5
    if x5 == 0:
        qc.rx((a1*pi) / 2, 4)
        qc.ry((a2*pi) / 32, 4)
        qc.rz((a3*pi) / 32, 4)
    else:
        qc.rx((a4*pi) / 2, 4)
        qc.ry((a5*pi) / 32, 4)
        qc.rz((a6*pi) / 32, 4)


    qc.barrier()
    for i in range (0, N_qubits):
        qc.measure(i, i)

    return qc

sampler = Sampler()

def quantum_strategy(x1, x2, x3, x4, x5):
    # `shots=1` runs the circuit once
    result = sampler.run(chsh_circuit(x1, x2, x3, x4, x5), shots=1).result()
    statistics = result.quasi_dists[0].binary_probabilities()
    bits = list(statistics.keys())[0]
    a1, a2, a3, a4, a5 = bits[0], bits[1], bits[2], bits[3], bits[4]
    return a1, a2, a3, a4, a5

def decide_the_classical_strategy(i1, i2, i3, i4, i5):
